def create
  if params[:shelf_name].size > 40 || params[:shelf_name].size < 1
    flash[:message] = "本棚名は40文字以内でお願いします。"
    render 'new' and return
  end

  group_id = session[:group_id]
  shelf = Shelf.find_all_by_group_id(group_id)
  if !shelf.empty?
    shelf_id = Shelf.find_all_by_group_id(group_id).last.shelf_id.to_i + 1
  else
    shelf_id = 1
  end

  shelf_name = params[:shelf_name]

  ActiveRecord::Base.transaction do
    shelf = Shelf.new(:shelf_id => shelf_id, :group_id => group_id, :name => shelf_name, :register_date => Time.now.strftime("%Y-%m-%d %H:%M:%S"))
    shelf.save!

    res = create_qr(shelf.shelf_id, group_id)
  end

  flash[:message] = "本棚「" + shelf_name + "」を登録しました。"
  @shelf = Shelf.find_by_group_id_and_shelf_id(group_id, shelf_id)
  @shelf_qr = "qr/qr_" + @shelf.group_id.to_s + "_" + @shelf.shelf_id.to_s
  render :action => 'show', :id => shelf_id
end

def create_qr(shelf_id, group_id)
  shelf = Shelf.find_by_shelf_id_and_group_id(shelf_id, group_id)
  group_name = Group.find(group_id).name

  # QRコード自体の大きさ
  width = 50
  height = 50

  # フォントサイズ
  fontsize = 20

  # 文字の位置
  group_x = width + 5
  shelf_x = group_x + group_name.size.to_i * fontsize + 10

  # ファイル名
  png_name = @qr_path + "qr_" + group_id.to_s + "_" + shelf_id.to_s + ".png"
  raw_png_name = @qr_path + "qr_" + group_id.to_s + "_" + shelf_id.to_s + "_raw.png"


  # QRコード画像の生成
  qr = RQRCode::QRCode.new(shelf_id.to_s, :size => 1, :level => :h )
  png = qr.to_img
  png.resize(width, height).save(raw_png_name)

  # 文字のサイズ分を右に伸ばす
  width += (group_name.size.to_i + shelf.name.size.to_i ) * 2 * fontsize
  logger.debug(group_name.size.to_s)
  logger.debug(shelf.name.size.to_s)

  # 文字表示の部分を拡張
  image = Magick::Image.read(raw_png_name).first
  image = image.change_geometry("#{width}x#{height}") do |cols,rows,img|
    img.resize!(cols, rows)
    img.background_color = 'white'
    img.extent(width, height)
  end

  # グループ名の表示
  draw = Magick::Draw.new
  draw.annotate(image, 0, 0, group_x , height / 2, group_name) do
    self.fill = '#2C7CFF'
    self.align = Magick::LeftAlign
    self.stroke = 'transparent'
    self.pointsize = fontsize
    self.text_antialias = true
    self.kerning = 1
    self.font = "./app/assets/fonts/ipaexg.ttf"
  end

  # 本棚名の表示
  draw.annotate(image, 0, 0, shelf_x , height / 2, shelf.name) do
    self.fill = '#FF367F'
    self.align = Magick::LeftAlign
    self.stroke = 'transparent'
    self.pointsize = fontsize
    self.text_antialias = true
    self.kerning = 1
    self.font = "./app/assets/fonts/ipaexg.ttf"
  end

  image.write(png_name)
end

def index
  @shelves = Shelf.find_all_by_group_id(session[:group_id])
  @each_shelf_book_num = {}
  @shelves.each do |shelf|
    group_books = Book.find_all_by_group_id_and_shelf_id(session[:group_id], shelf.shelf_id)
    @each_shelf_book_num["#{shelf.shelf_id}"] = group_books.size
  end
end
