def get_group_id(group_id, user_id)
  # パラメータにgroup_idがあった場合
  if group_id
    return group_id

  # いずれかのグループに所属していた場合
  elsif member = Member.find_by_user_id(user_id)
    return member.group_id
  else
    return nil
  end
end
