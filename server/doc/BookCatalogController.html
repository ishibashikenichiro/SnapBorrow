def create
  # 図書カタログがあるかどうか
  check_empty = BookCatalog.find_by_book_code(params[:book_code])

  # 図書カタログがないならば登録
  if check_empty.nil?
    Amazon::Ecs.options = {
      :associate_tag => ENV["ASSOCIATE_TAG"],
      :AWS_access_key_id => ENV["AWS_ACCESS_KEY_ID"],
      :AWS_secret_key => ENV["AWS_SECRET_KEY"]
    }

    # Amazonから図書情報を取得
    res = Amazon::Ecs.item_search(params[:book_code], {:search_index => 'Books', :response_group => 'Medium', :country => 'jp'})

    title = ""
    author = ""
    publisher = ""
    pub_date = ""

    # 取得した情報からDBに入れる情報を取得
    res.items.each do |item|
      title = (item.get_element("Title").to_s).gsub(/<.*?>/, "").encode("UTF-8")
      author = (item.get_element("Author").to_s).gsub(/<.*?>/, "").encode("UTF-8")
      publisher = (item.get_element("Publisher").to_s).gsub(/<.*?>/, "").encode("UTF-8")
      pub_date = (item.get_element("PublicationDate").to_s).gsub(/<.*?>/, "").encode("UTF-8")
    end

    # 出版日のフォーマットが良くない場合は1000-01-01にする
    if !valid_date(pub_date)
      pub_date = "1000-01-01"
    end

    # 図書カタログの登録
    ActiveRecord::Base.transaction do
      bc = BookCatalog.new(:book_code => params[:book_code],
                           :title => title,
                           :authors =>author,
                           :publisher => publisher,
                           :publication_date => pub_date)
      bc.save!
    end
  end
  render :nothing => true
end
