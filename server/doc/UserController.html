def create
  flash[:invite] = flash[:invite]
  flash[:url_hash] = flash[:url_hash]

  @user = User.new(:login_id => params[:login_id],
                   :nickname => params[:nickname],
#                     :mail => params[:mail],
                   :password => params[:password],
                   :register_date => Time.now.strftime("%Y-%m-%d %H:%M:%S"))

  if @user.save
    flash[:message] = "ユーザ登録が完了しました。右上のLoginからログインしてください。"
    session[:group_id] = 0
    if flash[:url_hash] && flash[:invite]
      session[:user_id] = @user.id
      redirect_to :controller =>'member', :action => 'invite_confirmation' and return
    end

    render 'success' and return
  else
    flash[:message] = "ユーザ登録ができませんでした。"
    render 'failure' and return
  end
end

def edit_profile
  @user = User.find_by_id(session[:user_id])
  if params[:password] != params[:password_confirmation]
    flash[:error] = 4
    render 'profile' and return
  end

  if (params[:password] == "" ||
      params[:password_confirmation] == "")
    flash[:error] = 5
    render 'profile' and return
  end


  @user.nickname = params[:nickname]
#    @user.mail = params[:mail]
  @user.password = params[:password]

  if @user.save
    flash[:message] = "プロフィールを更新しました。"
    render 'success' and return
  else
    flash[:message] = "プロフィールを更新できませんでした。"
    render 'failure' and return
  end
end
