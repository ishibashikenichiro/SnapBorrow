def create
  # 複数のbook_codeを取得
  book_codes = params[:book_code]
  book_count = 0
  shelf_name = ""

  # グループIDの取得
  group_id = get_group_id(params[:group_id], params[:user_id])

  # 所属しているグループが無い場合
  if !group_id
    render :json => regist_json(params[:qr], 0)
    return
  end

  # 本棚が登録されていなければshelf_idを0に
  if qr = Shelf.find_by_shelf_id_and_group_id(params[:qr], group_id)
    qr = qr.shelf_id
    shelf_name = Shelf.find_by_shelf_id_and_group_id(qr,group_id).name
  else
    qr = 0
    shelf_name = ""
  end

  # book_codeの数だけ実行
  book_codes.each do |book_code|
    # 図書カタログが登録されていなければスキップ
    bc = BookCatalog.find_by_book_code(book_code)
    if bc.nil?
      next
    end

    ActiveRecord::Base.transaction do
      # 図書の登録
      book = Book.new(:group_id => group_id, :book_code => book_code, :shelf_id => qr, :register_date => Time.now.strftime("%Y-%m-%d %H:%M:%S"))
      book.save!
      book_count += 1
    end
  end
  render :json => regist_json(shelf_name, book_count)
end

def regist_json(qr, book_num)
  json_array = {}
  result_array = Array.new
  result_array[0] = {"qr" => "#{qr}"}
  result_array[1] = {"book_num" => "#{book_num}"}
  json_array["result"] = result_array
  result = JSON.generate(json_array)
  return result
end

def search
  keyword = params[:keyword].strip
  user_id = session[:user_id]
  group_id = session[:group_id]

  @result = []
  if keyword == ""
    render 'search'
    return
  end

  logger.debug("keyword = 「" + keyword.to_s + "」")
  logger.debug("user_id = " + user_id.to_s)

  group_ids = Member.find_all_by_user_id(user_id)

  logger.debug("group_ids = " + group_ids.to_s)

  books_all = Book.arel_table
  @books = ""
  @books = books_all[:group_id].eq(group_ids.pop.group_id)
  group_ids.each do |group|
    @books = @books.or(books_all[:group_id].eq(group.group_id))
  end
  @books = Book.where(@books)
  logger.debug("books = " + @books.to_s)
  logger.debug("books = " + @books.to_sql)

  catalogs = BookCatalog.search(:title_or_authors_or_publisher_cont => StringEscape.new.escape_like(keyword)).result
  logger.debug("catalogs = " + catalogs.to_sql)

  catalogs_code = []
  catalogs.each do |catalog|
    catalogs_code.push(catalog.book_code)
  end
  @result = @books.search(:book_code_cont_any => catalogs_code).result
  if @result.to_sql == @books.to_sql
    @result = []
  end

  logger.debug("result = " + @result.to_s)

  flash[:keyword] = keyword
  render 'search'
end

def update_shelf_id
  book_code = params[:book_code]
  group_id = params[:group_id]
  shelf_id = params[:shelf_id]

  @other_log = nil
  books = Book.find_all_by_book_code_and_group_id_and_delete_date(book_code, group_id, nil)
  books.each do |book|
    @other_log = LendLog.find_by_book_id_and_temporary(book.id, true)
    if @other_log
      break
    end
  end

  @other_log.temporary = false
  @other_log.save!
  book = Book.find(@other_log.book_id)
  book.shelf_id = shelf_id
  book.save!

  render :json => LendLogController.new.array_to_json(0, book_code)
end
